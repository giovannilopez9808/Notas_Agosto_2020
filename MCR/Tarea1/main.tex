\begin{frame}
\frametitle{Tarea N$^{\circ}1$}
Ejericio: Mostrar que:a
\begin{align*}
    i) & \gamma \beta = Sinh \varsigma \\
    ii) & \gamma = Cosh  \varsigma \\
    iii) & A_0'B_0'-\vec{A}'\cdot \vec{B}'= A_0B_0-\vec{A}\cdot \vec{B}
\end{align*}
\end{frame}
\begin{frame}{i) $\gamma \beta = Sinh \varsigma $}
Sabemos que:
\begin{eqnarray}
\label{beta}
\beta=tanh \varsigma. \\
\gamma = \frac{1}{\sqrt{1-\beta^2}}.
\label{gamma}
\end{eqnarray}
\end{frame}
\begin{frame}{i) $\gamma\beta =Sinh \varsigma$}
Por lo que:
\begin{align*}
    \gamma \beta &= \frac{\beta}{\sqrt{1-\beta^2}}\\
    & = \frac{tanh \varsigma}{\sqrt{1-\varsigma^2}}\\
    & =\frac{tanh \varsigma}{\sqrt{sech^2 \varsigma}}\\
    & = \frac{tanh \varsigma}{sech \varsigma} \\
    & = sinh \varepsilon
\end{align*}
\end{frame}
\begin{frame}{i) $\gamma \beta = sinh \varsigma$}
    por lo tanto 
    \begin{equation}
        \gamma \beta = sinh \varsigma
    \end{equation}
\end{frame}
